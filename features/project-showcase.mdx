## Overview

The project showcase feature allows you to display your personal projects alongside your on-chain work experiences. Unlike experiences, projects are currently stored in browser **localStorage** for instant updates without gas fees.

<Warning>
  Projects are stored **off-chain** in your browser's localStorage. This means they are device-specific and will be lost if you clear your browser data.
</Warning>

## Adding Projects

Projects are added through a dedicated form at `/form/projects`:

### Project Form Interface

```tsx src/app/form/projects/page.tsx
import { useState, useEffect } from 'react';
import { useRouter } from 'next/navigation';

export default function ProjectsForm() {
  const router = useRouter();
  const [projects, setProjects] = useState([
    { name: '', link: '', description: '' }
  ]);

  useEffect(() => {
    const savedData = localStorage.getItem('portfolioProjects');
    if (savedData) {
      setProjects(JSON.parse(savedData));
    }
  }, []);

  const handleProjectChange = (index: number, field: string, value: string) => {
    const newProjects = [...projects];
    newProjects[index] = { ...newProjects[index], [field]: value };
    setProjects(newProjects);
  };

  const addProject = () => {
    setProjects([...projects, { name: '', link: '', description: '' }]);
  };

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    localStorage.setItem('portfolioProjects', JSON.stringify(projects));
    router.push('/portfolio');
  };

  return (
    <form onSubmit={handleSubmit}>
      {projects.map((project, index) => (
        <div key={index}>
          <Input
            value={project.name}
            onChange={(e) => handleProjectChange(index, 'name', e.target.value)}
            placeholder="Project Name"
          />
          <Input
            value={project.link}
            onChange={(e) => handleProjectChange(index, 'link', e.target.value)}
            placeholder="Project Link"
          />
          <Textarea
            value={project.description}
            onChange={(e) => handleProjectChange(index, 'description', e.target.value)}
            placeholder="Project Description"
          />
        </div>
      ))}
      <Button type="button" onClick={addProject}>
        Add Project
      </Button>
      <Button type="submit">
        Save and View Portfolio
      </Button>
    </form>
  );
}
```

## LocalStorage Integration

Projects are stored in browser localStorage under the key `portfolioProjects`:

### Saving Projects

```typescript
const handleSubmit = (e: React.FormEvent) => {
  e.preventDefault();
  localStorage.setItem('portfolioProjects', JSON.stringify(projects));
  router.push('/portfolio');
};
```

### Loading Projects

The form automatically loads previously saved projects:

```typescript
useEffect(() => {
  const savedData = localStorage.getItem('portfolioProjects');
  if (savedData) {
    setProjects(JSON.parse(savedData));
  }
}, []);
```

<Note>

</Note>

### LocalStorage vs Blockchain

<Tabs>
  <Tab title="localStorage (Projects)">
    ```javascript
    // No wallet needed
    localStorage.setItem('portfolioProjects', JSON.stringify(projects));
    
    // Instant, no transaction
    const projects = JSON.parse(localStorage.getItem('portfolioProjects'));
    
    // Can edit/delete freely
    localStorage.removeItem('portfolioProjects');
    ```
    
    - Free and instant
    - Easy to edit
    - Browser-dependent
  </Tab>
  
  <Tab title="Blockchain (Experiences)">
    ```javascript
    // Requires wallet
    const { writeContract } = useWriteContract();
    
    // Transaction required (gas fees)
    writeContract({
      address: CONTRACT_ADDRESS,
      functionName: "addWork",
      args: [role, description, link, company]
    });
    
    // Immutable - cannot edit
    const { data } = useReadContract({
      functionName: "getAllWork"
    });
    ```
    
    - Requires gas fees
    - Immutable storage
    - Portable across devices
  </Tab>
</Tabs>

## Complete Example

```typescript
import { useState, useEffect } from 'react';
import { useRouter } from 'next/navigation';

interface Project {
  name: string;
  link: string;
  description: string;
}

export default function ProjectsForm() {
  const router = useRouter();
  const [projects, setProjects] = useState<Project[]>([
    { name: '', link: '', description: '' }
  ]);

  // Load existing projects
  useEffect(() => {
    const saved = localStorage.getItem('portfolioProjects');
    if (saved) {
      setProjects(JSON.parse(saved));
    }
  }, []);

  // Add new project entry
  const addProject = () => {
    setProjects([...projects, { name: '', link: '', description: '' }]);
  };

  // Update project field
  const updateProject = (index: number, field: keyof Project, value: string) => {
    const updated = [...projects];
    updated[index] = { ...updated[index], [field]: value };
    setProjects(updated);
  };

  // Save and redirect
  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    localStorage.setItem('portfolioProjects', JSON.stringify(projects));
    router.push('/portfolio');
  };

  return (
    <form onSubmit={handleSubmit}>
      {projects.map((project, i) => (
        <div key={i}>
          <input
            value={project.name}
            onChange={(e) => updateProject(i, 'name', e.target.value)}
            placeholder="Project Name"
          />
          <input
            value={project.link}
            onChange={(e) => updateProject(i, 'link', e.target.value)}
            placeholder="Project Link"
          />
          <textarea
            value={project.description}
            onChange={(e) => updateProject(i, 'description', e.target.value)}
            placeholder="Description"
          />
        </div>
      ))}
      <button type="button" onClick={addProject}>Add Project</button>
      <button type="submit">Save</button>
    </form>
  );
}
```
